304 // A queue of objects promoted during scavenge. Each object is accompanied
305 // by it's size to avoid dereferencing a map pointer for scanning.
306 // The last page in to-space is used for the promotion queue. On conflict
307 // during scavenge, the promotion queue is allocated externally and all
308 // entries are copied to the external queue.
309 class PromotionQueue {
310  public:
311  explicit PromotionQueue(Heap* heap)
312  : front_(NULL),
313  rear_(NULL),
314  limit_(NULL),
315  emergency_stack_(0),
316  heap_(heap) {}
317 
318  void Initialize();
319 
320  void Destroy() {
321  DCHECK(is_empty());
322  delete emergency_stack_;
323  emergency_stack_ = NULL;
324  }
325 
326  Page* GetHeadPage() {
327  return Page::FromAllocationAreaAddress(reinterpret_cast<Address>(rear_));
328  }
329 
330  void SetNewLimit(Address limit) {
331  // If we are already using an emergency stack, we can ignore it.
332  if (emergency_stack_) return;
333 
334  // If the limit is not on the same page, we can ignore it.
335  if (Page::FromAllocationAreaAddress(limit) != GetHeadPage()) return;
336 
337  limit_ = reinterpret_cast<struct Entry*>(limit);
338 
339  if (limit_ <= rear_) {
340  return;
341  }
342 
343  RelocateQueueHead();
344  }
345 
346  bool IsBelowPromotionQueue(Address to_space_top) {
347  // If an emergency stack is used, the to-space address cannot interfere
348  // with the promotion queue.
349  if (emergency_stack_) return true;
350 
351  // If the given to-space top pointer and the head of the promotion queue
352  // are not on the same page, then the to-space objects are below the
353  // promotion queue.
354  if (GetHeadPage() != Page::FromAddress(to_space_top)) {
355  return true;
356  }
357  // If the to space top pointer is smaller or equal than the promotion
358  // queue head, then the to-space objects are below the promotion queue.
359  return reinterpret_cast<struct Entry*>(to_space_top) <= rear_;
360  }
361 
362  bool is_empty() {
363  return (front_ == rear_) &&
364  (emergency_stack_ == NULL || emergency_stack_->length() == 0);
365  }
366 
367  inline void insert(HeapObject* target, int32_t size, bool was_marked_black);
368 
369  void remove(HeapObject** target, int32_t* size, bool* was_marked_black) {
370  DCHECK(!is_empty());
371  if (front_ == rear_) {
372  Entry e = emergency_stack_->RemoveLast();
373  *target = e.obj_;
374  *size = e.size_;
375  *was_marked_black = e.was_marked_black_;
376  return;
377  }
378 
379  struct Entry* entry = reinterpret_cast<struct Entry*>(--front_);
380  *target = entry->obj_;
381  *size = entry->size_;
382  *was_marked_black = entry->was_marked_black_;
383 
384  // Assert no underflow.
385  SemiSpace::AssertValidRange(reinterpret_cast<Address>(rear_),
386  reinterpret_cast<Address>(front_));
387  }
388 
389  private:
390  struct Entry {
391  Entry(HeapObject* obj, int32_t size, bool was_marked_black)
392  : obj_(obj), size_(size), was_marked_black_(was_marked_black) {}
393 
394  HeapObject* obj_;
395  int32_t size_ : 31;
396  bool was_marked_black_ : 1;
397  };

2607  // more frequently than the step_size we are monitoring (e.g. when there are
2608  // multiple observers, or when page or space boundary is encountered.)
2609  void AllocationStep(int bytes_allocated, Address soon_object, size_t size) {
2610  bytes_to_next_step_ -= bytes_allocated;
2611  if (bytes_to_next_step_ <= 0) {
2612  Step(static_cast<int>(step_size_ - bytes_to_next_step_), soon_object,
2613  size);
2614  step_size_ = GetNextStepSize();
2615  bytes_to_next_step_ = step_size_;

2621  intptr_t bytes_to_next_step() const { return bytes_to_next_step_; }

2624  // least step_size bytes have been allocated. soon_object is the address just
2625  // allocated (but not yet initialized.) size is the size of the object as
2626  // requested (i.e. w/o the alignment fillers). Some complexities to be aware
2627  // of:
2628  // 1) soon_object will be nullptr in cases where we end up observing an
2629  // allocation that happens to be a filler space (e.g. page boundaries.)
2630  // 2) size is the requested size at the time of allocation. Right-trimming
2631  // may change the object size dynamically.

2633  // group. In such a case size is the size of the group rather than the
2634  // first object.
2635  virtual void Step(int bytes_allocated, Address soon_object, size_t size) = 0;

2638  virtual intptr_t GetNextStepSize() { return step_size_; }
