189  // Get a string stored in the module bytes representing a name.
190  WasmName GetName(uint32_t offset, uint32_t length) const {
191  if (length == 0) return {"<?>", 3}; // no name.
192  CHECK(BoundsCheck(offset, offset + length));
193  DCHECK_GE(static_cast<int>(length), 0);
