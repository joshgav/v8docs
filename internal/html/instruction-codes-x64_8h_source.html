149 // Addressing modes represent the "shape" of inputs to an instruction.
150 // Many instructions support multiple addressing modes. Addressing modes
151 // are encoded into the InstructionCode of the instruction and tell the
152 // code generator after register allocation which assembler method to call.
153 //
154 // We use the following local notation for addressing modes:
155 //
156 // M = memory operand
157 // R = base register
158 // N = index register * N for N in {1, 2, 4, 8}
159 // I = immediate displacement (32-bit signed integer)
160 
161 #define TARGET_ADDRESSING_MODE_LIST(V) \
162  V(MR) /* [%r1 ] */ \
163  V(MRI) /* [%r1 + K] */ \
164  V(MR1) /* [%r1 + %r2*1 ] */ \
165  V(MR2) /* [%r1 + %r2*2 ] */ \
166  V(MR4) /* [%r1 + %r2*4 ] */ \
167  V(MR8) /* [%r1 + %r2*8 ] */ \
168  V(MR1I) /* [%r1 + %r2*1 + K] */ \
169  V(MR2I) /* [%r1 + %r2*2 + K] */ \
170  V(MR4I) /* [%r1 + %r2*3 + K] */ \
171  V(MR8I) /* [%r1 + %r2*4 + K] */ \
172  V(M1) /* [ %r2*1 ] */ \
173  V(M2) /* [ %r2*2 ] */ \
174  V(M4) /* [ %r2*4 ] */ \
175  V(M8) /* [ %r2*8 ] */ \
176  V(M1I) /* [ %r2*1 + K] */ \
177  V(M2I) /* [ %r2*2 + K] */ \
178  V(M4I) /* [ %r2*4 + K] */ \
179  V(M8I) /* [ %r2*8 + K] */
