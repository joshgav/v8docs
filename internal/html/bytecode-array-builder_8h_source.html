388 // A label representing a branch target in a bytecode array. When a
389 // label is bound, it represents a known position in the bytecode
390 // array. For labels that are forward references there can be at most
391 // one reference whilst it is unbound.
392 class BytecodeLabel final {
393  public:
394  BytecodeLabel() : bound_(false), offset_(kInvalidOffset) {}
395 
396  bool is_bound() const { return bound_; }
397  size_t offset() const { return offset_; }
398 
399  private:
400  static const size_t kInvalidOffset = static_cast<size_t>(-1);
401 
402  void bind_to(size_t offset) {
403  DCHECK(!bound_ && offset != kInvalidOffset);
404  offset_ = offset;
405  bound_ = true;
406  }
407 
408  void set_referrer(size_t offset) {
409  DCHECK(!bound_ && offset != kInvalidOffset && offset_ == kInvalidOffset);
410  offset_ = offset;
411  }
412 
413  bool is_forward_target() const {
414  return offset() != kInvalidOffset && !is_bound();
415  }
416 
417  // There are three states for a label:
418  // bound_ offset_
419  // UNSET false kInvalidOffset
420  // FORWARD_TARGET false Offset of referring jump
421  // BACKWARD_TARGET true Offset of label in bytecode array when bound
422  bool bound_;
423  size_t offset_;
424 
425  friend class BytecodeArrayBuilder;
426 };
