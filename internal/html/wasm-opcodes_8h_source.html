452  static WasmOpcode LoadStoreOpcodeOf(MachineType type, bool store) {
453  if (type == MachineType::Int8()) {
454  return store ? kExprI32StoreMem8 : kExprI32LoadMem8S;

456  return store ? kExprI32StoreMem8 : kExprI32LoadMem8U;
457  } else if (type == MachineType::Int16()) {
458  return store ? kExprI32StoreMem16 : kExprI32LoadMem16S;
459  } else if (type == MachineType::Uint16()) {
460  return store ? kExprI32StoreMem16 : kExprI32LoadMem16U;
461  } else if (type == MachineType::Int32()) {
462  return store ? kExprI32StoreMem : kExprI32LoadMem;
463  } else if (type == MachineType::Uint32()) {
464  return store ? kExprI32StoreMem : kExprI32LoadMem;
465  } else if (type == MachineType::Int64()) {
