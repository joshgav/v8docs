153 struct BranchTableOperand {
154  uint32_t arity;
155  uint32_t table_count;
156  const byte* table;
157  int length;
158  inline BranchTableOperand(Decoder* decoder, const byte* pc) {
159  int len1 = 0;
160  int len2 = 0;
161  arity = decoder->checked_read_u32v(pc, 1, &len1, "argument count");
162  table_count =
163  decoder->checked_read_u32v(pc, 1 + len1, &len2, "table count");
164  length = len1 + len2 + (table_count + 1) * sizeof(uint32_t);
165 
166  uint32_t table_start = 1 + len1 + len2;
167  if (decoder->check(pc, table_start, (table_count + 1) * sizeof(uint32_t),
168  "expected <table entries>")) {
169  table = pc + table_start;
170  } else {
171  table = nullptr;
172  }
173  }
174  inline uint32_t read_entry(Decoder* decoder, int i) {
175  DCHECK(i >= 0 && static_cast<uint32_t>(i) <= table_count);
176  return table ? decoder->read_u32(table + i * sizeof(uint32_t)) : 0;
177  }
178 };
