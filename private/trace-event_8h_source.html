21 // The pointer returned from GetCategoryGroupEnabled() points to a

31  // Category group enabled to export events to ETW.

124 // Implementation detail: internal macro to create static category.
125 // No barriers are needed, because this code is designed to operate safely
126 // even when the unsigned char* points to garbage data (which may be the case
127 // on processors without cache coherency).
128 // TODO(fmeawad): This implementation contradicts that we can have a different
129 // configuration for each isolate,

131 #define INTERNAL_TRACE_EVENT_GET_CATEGORY_INFO_CUSTOM_VARIABLES( \
132  category_group, atomic, category_group_enabled) \
133  category_group_enabled = \
134  reinterpret_cast<const uint8_t*>(TRACE_EVENT_API_ATOMIC_LOAD(atomic)); \
135  if (!category_group_enabled) { \

137  TRACE_EVENT_API_GET_CATEGORY_GROUP_ENABLED(category_group); \
138  TRACE_EVENT_API_ATOMIC_STORE( \

424 static V8_INLINE uint64_t AddTraceEventImpl(
425  char phase, const uint8_t* category_group_enabled, const char* name,
426  const char* scope, uint64_t id, uint64_t bind_id, int32_t num_args,
427  const char** arg_names, const uint8_t* arg_types,
428  const uint64_t* arg_values, unsigned int flags) {
429  std::unique_ptr<ConvertableToTraceFormat> arg_convertables[2];
430  if (num_args > 0 && arg_types[0] == TRACE_VALUE_TYPE_CONVERTABLE) {
431  arg_convertables[0].reset(reinterpret_cast<ConvertableToTraceFormat*>(
432  static_cast<intptr_t>(arg_values[0])));
433  }
434  if (num_args > 1 && arg_types[1] == TRACE_VALUE_TYPE_CONVERTABLE) {
435  arg_convertables[1].reset(reinterpret_cast<ConvertableToTraceFormat*>(
436  static_cast<intptr_t>(arg_values[1])));
437  }
438  DCHECK(num_args <= 2);
439  v8::Platform* platform =
440  v8::internal::tracing::TraceEventHelper::GetCurrentPlatform();
441  return platform->AddTraceEvent(phase, category_group_enabled, name, scope, id,
442  bind_id, num_args, arg_names, arg_types,
443  arg_values, arg_convertables, flags);
444 }
