29 inline BranchHint NegateBranchHint(BranchHint hint) {
30  switch (hint) {
31  case BranchHint::kNone:
32  return hint;
33  case BranchHint::kTrue:
34  return BranchHint::kFalse;
35  case BranchHint::kFalse:
36  return BranchHint::kTrue;
37  }
38  UNREACHABLE();
39  return hint;
