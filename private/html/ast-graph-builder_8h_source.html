408 // The abstract execution environment for generated code consists of
409 // parameter variables, local variables and the operand stack. The
410 // environment will perform proper SSA-renaming of all tracked nodes
411 // at split and merge points in the control flow. Internally all the
412 // values are stored in one list using the following layout:
413 //
414 // [parameters (+receiver)] [locals] [operand stack]
415 //
416 class AstGraphBuilder::Environment : public ZoneObject {
417  public:
418  Environment(AstGraphBuilder* builder, DeclarationScope* scope,
419  Node* control_dependency);
420 
421  int parameters_count() const { return parameters_count_; }
422  int locals_count() const { return locals_count_; }
423  int context_chain_length() { return static_cast<int>(contexts_.size()); }
424  int stack_height() {
425  return static_cast<int>(values()->size()) - parameters_count_ -
426  locals_count_;
427  }
428 
429  // Operations on parameter or local variables.
430  void Bind(Variable* variable, Node* node);
431  Node* Lookup(Variable* variable);
432  void MarkAllLocalsLive();
433 
434  // Raw operations on parameter variables.
435  void RawParameterBind(int index, Node* node);
436  Node* RawParameterLookup(int index);
437 
438  // Operations on the context chain.
439  Node* Context() const { return contexts_.back(); }
440  void PushContext(Node* context) { contexts()->push_back(context); }
441  void PopContext() { contexts()->pop_back(); }
442  void TrimContextChain(int trim_to_length) {
443  contexts()->resize(trim_to_length);
444  }
445 
446  // Operations on the operand stack.
447  void Push(Node* node) {
448  values()->push_back(node);
449  }
450  Node* Top() {
451  DCHECK(stack_height() > 0);
452  return values()->back();
453  }
454  Node* Pop() {
455  DCHECK(stack_height() > 0);
456  Node* back = values()->back();
457  values()->pop_back();
458  return back;
459  }
460 
461  // Direct mutations of the operand stack.
462  void Poke(int depth, Node* node) {
463  DCHECK(depth >= 0 && depth < stack_height());
464  int index = static_cast<int>(values()->size()) - depth - 1;
465  values()->at(index) = node;
466  }
467  Node* Peek(int depth) {
468  DCHECK(depth >= 0 && depth < stack_height());
469  int index = static_cast<int>(values()->size()) - depth - 1;
470  return values()->at(index);
471  }
472  void Drop(int depth) {
473  DCHECK(depth >= 0 && depth <= stack_height());
474  values()->erase(values()->end() - depth, values()->end());
475  }
476  void TrimStack(int trim_to_height) {
477  int depth = stack_height() - trim_to_height;
478  DCHECK(depth >= 0 && depth <= stack_height());
479  values()->erase(values()->end() - depth, values()->end());
480  }
481

483  // further mutation of the environment will not affect checkpoints.
484  Node* Checkpoint(BailoutId ast_id, OutputFrameStateCombine combine =

486  bool node_has_exception = false);

491 
492  // Control dependency tracked by this environment.
