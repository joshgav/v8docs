76  // StopUsing() is called.
77  void StartUsing();
78  void StopUsing();
79  void Clear();
80 
81  inline bool IsFull() { return ((top_ + 1) & mask_) == bottom_; }
82 
83  inline bool IsEmpty() { return top_ == bottom_; }
84 
85  bool overflowed() const { return overflowed_; }
86 
87  void ClearOverflowed() { overflowed_ = false; }
88 
89  void SetOverflowed() { overflowed_ = true; }
90 
91  // Push the object on the marking stack if there is room, otherwise mark the
92  // deque as overflowed and wait for a rescan of the heap.
93  INLINE(bool Push(HeapObject* object)) {
94  DCHECK(object->IsHeapObject());
95  if (IsFull()) {
96  SetOverflowed();
97  return false;
98  } else {
99  array_[top_] = object;
100  top_ = ((top_ + 1) & mask_);
101  return true;
102  }
103  }
104 
105  INLINE(HeapObject* Pop()) {
106  DCHECK(!IsEmpty());
107  top_ = ((top_ - 1) & mask_);
108  HeapObject* object = array_[top_];
109  DCHECK(object->IsHeapObject());
110  return object;
111  }
112 
113  // Unshift the object into the marking stack if there is room, otherwise mark
114  // the deque as overflowed and wait for a rescan of the heap.
115  INLINE(bool Unshift(HeapObject* object)) {
116  DCHECK(object->IsHeapObject());
117  if (IsFull()) {
118  SetOverflowed();
119  return false;
120  } else {
121  bottom_ = ((bottom_ - 1) & mask_);
122  array_[bottom_] = object;
123  return true;
124  }
125  }
126 
127  HeapObject** array() { return array_; }
128  int bottom() { return bottom_; }
129  int top() { return top_; }
130  int mask() { return mask_; }
131  void set_top(int top) { top_ = top; }
132

246 class MarkBitCellIterator BASE_EMBEDDED {
247  public:
248  explicit MarkBitCellIterator(MemoryChunk* chunk) : chunk_(chunk) {
249  last_cell_index_ = Bitmap::IndexToCell(Bitmap::CellAlignIndex(
250  chunk_->AddressToMarkbitIndex(chunk_->area_end())));
251  cell_base_ = chunk_->area_start();
252  cell_index_ = Bitmap::IndexToCell(
253  Bitmap::CellAlignIndex(chunk_->AddressToMarkbitIndex(cell_base_)));
254  cells_ = chunk_->markbits()->cells();
255  }
256 
257  inline bool Done() { return cell_index_ == last_cell_index_; }
258 
259  inline bool HasNext() { return cell_index_ < last_cell_index_ - 1; }
260 
261  inline MarkBit::CellType* CurrentCell() {
262  DCHECK(cell_index_ == Bitmap::IndexToCell(Bitmap::CellAlignIndex(
263  chunk_->AddressToMarkbitIndex(cell_base_))));
264  return &cells_[cell_index_];
265  }
266 
267  inline Address CurrentCellBase() {
268  DCHECK(cell_index_ == Bitmap::IndexToCell(Bitmap::CellAlignIndex(
269  chunk_->AddressToMarkbitIndex(cell_base_))));
270  return cell_base_;
271  }
272 
273  inline void Advance() {
274  cell_index_++;
275  cell_base_ += Bitmap::kBitsPerCell * kPointerSize;
276  }
277 
278  inline bool Advance(unsigned int new_cell_index) {
279  if (new_cell_index != cell_index_) {
280  DCHECK_GT(new_cell_index, cell_index_);
281  DCHECK_LE(new_cell_index, last_cell_index_);
282  unsigned int diff = new_cell_index - cell_index_;
283  cell_index_ = new_cell_index;
284  cell_base_ += diff * (Bitmap::kBitsPerCell * kPointerSize);
285  return true;
286  }
287  return false;
288  }
289 
290  // Return the next mark bit cell. If there is no next it returns 0;
291  inline MarkBit::CellType PeekNext() {
292  if (HasNext()) {
293  return cells_[cell_index_ + 1];
294  }
295  return 0;
296  }
297 
298  private:
299  MemoryChunk* chunk_;
300  MarkBit::CellType* cells_;
301  unsigned int last_cell_index_;
302  unsigned int cell_index_;
303  Address cell_base_;
304 };
