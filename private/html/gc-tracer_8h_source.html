101 // GCTracer collects and prints ONE line after each garbage collector
102 // invocation IFF --trace_gc is used.
103 class V8_EXPORT_PRIVATE GCTracer {
104  public:
105  struct IncrementalMarkingInfos {
106  IncrementalMarkingInfos() : duration(0), longest_step(0), steps(0) {}
107 
108  void Update(double duration) {
109  steps++;
110  this->duration += duration;
111  if (duration > longest_step) {
112  longest_step = duration;
113  }
114  }
115 
116  void ResetCurrentCycle() {
117  duration = 0;
118  longest_step = 0;
119  steps = 0;
120  }
121 
122  double duration;
123  double longest_step;
124  int steps;
125  };

324  V8_INLINE void AddScopeSample(Scope::ScopeId scope, double duration) {
325  DCHECK(scope < Scope::NUMBER_OF_SCOPES);
326  if (scope >= Scope::FIRST_INCREMENTAL_SCOPE &&
327  scope <= Scope::LAST_INCREMENTAL_SCOPE) {
328  incremental_marking_scopes_[scope - Scope::FIRST_INCREMENTAL_SCOPE]
329  .Update(duration);
330  } else {
331  current_.scopes[scope] += duration;
332  }
333  }
