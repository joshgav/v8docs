28 // The V8 compiler API.
29 //
30 // This is the central hub for dispatching to the various compilers within V8.
31 // Logic for which compiler to choose and how to wire compilation results into
32 // the object heap should be kept inside this class.
33 //
34 // General strategy: Scripts are translated into anonymous functions w/o
35 // parameters which then can be executed. If the source code contains other
36 // functions, they might be compiled and allocated as part of the compilation
37 // of the source code or deferred for lazy compilation at a later point.
38 class Compiler : public AllStatic {
39  public:
40  enum ClearExceptionFlag { KEEP_EXCEPTION, CLEAR_EXCEPTION };
41  enum ConcurrencyMode { NOT_CONCURRENT, CONCURRENT };
42  enum CompilationTier { INTERPRETED, BASELINE, OPTIMIZED };

144 // A base class for compilation jobs intended to run concurrent to the main
145 // thread. The job is split into three phases which are called in sequence on
146 // different threads and with different limitations:
147 // 1) PrepareJob: Runs on main thread. No major limitations.
148 // 2) ExecuteJob: Runs concurrently. No heap allocation or handle derefs.
149 // 3) FinalizeJob: Runs on main thread. No dependency changes.
150 //
151 // Each of the three phases can either fail or succeed. The current state of
152 // the job can be checked using {state()}.
153 class CompilationJob {
154  public:
155  enum Status { SUCCEEDED, FAILED };
156  enum class State {
157  kReadyToPrepare,
158  kReadyToExecute,
159  kReadyToFinalize,
160  kSucceeded,
161  kFailed,
162  };
163 
164  CompilationJob(Isolate* isolate, CompilationInfo* info,
165  const char* compiler_name,
166  State initial_state = State::kReadyToPrepare);
167  virtual ~CompilationJob() {}
168 
169  // Prepare the compile job. Must be called on the main thread.
170  MUST_USE_RESULT Status PrepareJob();
171 
172  // Executes the compile job. Can be called on a background thread if
173  // can_execute_on_background_thread() returns true.
174  MUST_USE_RESULT Status ExecuteJob();
175 
176  // Finalizes the compile job. Must be called on the main thread.
177  MUST_USE_RESULT Status FinalizeJob();
178 
179  // Report a transient failure, try again next time. Should only be called on
180  // optimization compilation jobs.
181  Status RetryOptimization(BailoutReason reason);
182 
183  // Report a persistent failure, disable future optimization on the function.
184  // Should only be called on optimization compilation jobs.
185  Status AbortOptimization(BailoutReason reason);
186 
187  void RecordOptimizedCompilationStats() const;
188  void RecordUnoptimizedCompilationStats() const;
189 
190  virtual bool can_execute_on_background_thread() const { return true; }
191 
192  void set_stack_limit(uintptr_t stack_limit) { stack_limit_ = stack_limit; }
193  uintptr_t stack_limit() const { return stack_limit_; }
194 
195  bool executed_on_background_thread() const {
196  DCHECK_IMPLIES(!can_execute_on_background_thread(),
197  !executed_on_background_thread_);
198  return executed_on_background_thread_;
199  }
200  State state() const { return state_; }
201  CompilationInfo* info() const { return info_; }
202  Isolate* isolate() const;
203 
204  protected:
205  // Overridden by the actual implementation.
206  virtual Status PrepareJobImpl() = 0;
207  virtual Status ExecuteJobImpl() = 0;
208  virtual Status FinalizeJobImpl() = 0;
209 
210  // Registers weak object to optimized code dependencies.
211  // TODO(turbofan): Move this to pipeline.cc once Crankshaft dies.
212  void RegisterWeakObjectsInOptimizedCode(Handle<Code> code);
213 
214  private:
215  CompilationInfo* info_;
216  ThreadId isolate_thread_id_;
217  base::TimeDelta time_taken_to_prepare_;
218  base::TimeDelta time_taken_to_execute_;
219  base::TimeDelta time_taken_to_finalize_;
220  const char* compiler_name_;
221  State state_;
222  uintptr_t stack_limit_;
223  bool executed_on_background_thread_;
224 
225  MUST_USE_RESULT Status UpdateState(Status status, State next_state) {
226  if (status == SUCCEEDED) {
227  state_ = next_state;
228  } else {
229  state_ = State::kFailed;
230  }
231  return status;
232  }
233 };
