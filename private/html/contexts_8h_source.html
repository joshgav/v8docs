351 // A table of all script contexts. Every loaded top-level script with top-level
352 // lexical declarations contributes its ScriptContext into this table.
353 //
354 // The table is a fixed array, its first slot is the current used count and
355 // the subsequent slots 1..used contain ScriptContexts.
356 class ScriptContextTable : public FixedArray {
357  public:
358  // Conversions.
359  static inline ScriptContextTable* cast(Object* context);
360 
361  struct LookupResult {
362  int context_index;
363  int slot_index;
364  VariableMode mode;
365  InitializationFlag init_flag;
366  MaybeAssignedFlag maybe_assigned_flag;
367  };
368 
369  inline int used() const;
370  inline void set_used(int used);
371 
372  static inline Handle<Context> GetContext(Handle<ScriptContextTable> table,
373  int i);
374 
375  // Lookup a variable `name` in a ScriptContextTable.
376  // If it returns true, the variable is found and `result` contains
377  // valid information about its location.
378  // If it returns false, `result` is untouched.
379  MUST_USE_RESULT
380  static bool Lookup(Handle<ScriptContextTable> table, Handle<String> name,
381  LookupResult* result);
382 
383  MUST_USE_RESULT
384  static Handle<ScriptContextTable> Extend(Handle<ScriptContextTable> table,

400 // JSFunctions are pairs (context, function code), sometimes also called
401 // closures. A Context object is used to represent function contexts and
402 // dynamically pushed 'with' contexts (or 'scopes' in ECMA-262 speak).
403 //
404 // At runtime, the contexts build a stack in parallel to the execution
405 // stack, with the top-most context being the current context. All contexts
406 // have the following slots:
407 //
408 // [ closure ] This is the current function. It is the same for all
409 // contexts inside a function. It provides access to the
410 // incoming context (i.e., the outer context, which may
411 // or may not become the current function's context), and
412 // it provides access to the functions code and thus it's
413 // scope information, which in turn contains the names of
414 // statically allocated context slots. The names are needed
415 // for dynamic lookups in the presence of 'with' or 'eval'.
416 //
417 // [ previous ] A pointer to the previous context.
418 //
419 // [ extension ] Additional data.
420 //
421 // For script contexts, it contains the respective ScopeInfo.
422 //
423 // For catch contexts, it contains a ContextExtension object
424 // consisting of the ScopeInfo and the name of the catch
425 // variable.
426 //
427 // For module contexts, it contains the module object.
428 //
429 // For block contexts, it contains either the respective
430 // ScopeInfo or a ContextExtension object consisting of the
431 // ScopeInfo and an "extension object" (see below).
432 //
433 // For with contexts, it contains a ContextExtension object
434 // consisting of the ScopeInfo and an "extension object".
435 //
436 // An "extension object" is used to dynamically extend a context
437 // with additional variables, namely in the implementation of the
438 // 'with' construct and the 'eval' construct. For instance,
439 // Context::Lookup also searches the extension object for
440 // properties. (Storing the extension object is the original
441 // purpose of this context slot, hence the name.)
442 //
443 // [ native_context ] A pointer to the native context.
444 //
445 // In addition, function contexts may have statically allocated context slots
446 // to store local variables/functions that are accessed from inner functions
447 // (via static context addresses) or through 'eval' (dynamic context lookups).
448 // The native context contains additional slots for fast access to native
449 // properties.
450 //
451 // Finally, with Harmony scoping, the JSFunction representing a top level
452 // script will have the ScriptContext rather than a FunctionContext.
453 // Script contexts from all top-level scripts are gathered in
454 // ScriptContextTable.
