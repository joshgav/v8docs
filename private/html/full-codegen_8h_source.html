91  class NestedStatement BASE_EMBEDDED {
92  public:
93  explicit NestedStatement(FullCodeGenerator* codegen)
94  : codegen_(codegen),
95  stack_depth_at_target_(codegen->operand_stack_depth_) {
96  // Link into codegen's nesting stack.
97  previous_ = codegen->nesting_stack_;
98  codegen->nesting_stack_ = this;
99  }
100  virtual ~NestedStatement() {
101  // Unlink from codegen's nesting stack.
102  DCHECK_EQ(this, codegen_->nesting_stack_);
103  codegen_->nesting_stack_ = previous_;
104  }
105 
106  virtual Breakable* AsBreakable() { return nullptr; }
107  virtual Iteration* AsIteration() { return nullptr; }
108 
109  virtual bool IsContinueTarget(Statement* target) { return false; }
110  virtual bool IsBreakTarget(Statement* target) { return false; }
111 
112  // Notify the statement that we are exiting it via break, continue, or
113  // return and give it a chance to generate cleanup code. Return the
114  // next outer statement in the nesting stack. We accumulate in
115  // {*context_length} the number of context chain links to unwind as we
116  // traverse the nesting stack from an exit to its target.
117  virtual NestedStatement* Exit(int* context_length) { return previous_; }
118 
119  // Determine the expected operand stack depth when this statement is being
120  // used as the target of an exit. The caller will drop to this depth.
121  int GetStackDepthAtTarget() { return stack_depth_at_target_; }
122 
123  protected:
124  MacroAssembler* masm() { return codegen_->masm(); }
125 
126  FullCodeGenerator* codegen_;
127  NestedStatement* previous_;
128  int stack_depth_at_target_;
129 
130  private:
131  DISALLOW_COPY_AND_ASSIGN(NestedStatement);
