57 // Global invariants after AST construction: Each reference (i.e. identifier)
58 // to a JavaScript variable (including global properties) is represented by a
59 // VariableProxy node. Immediately after AST construction and before variable
60 // allocation, most VariableProxy nodes are "unresolved", i.e. not bound to a
61 // corresponding variable (though some are bound during parse time). Variable
62 // allocation binds each unresolved VariableProxy to one Variable and assigns
63 // a location. Note that many VariableProxy nodes may refer to the same Java-
64 // Script variable.
65 
66 // JS environments are represented in the parser using Scope, DeclarationScope
67 // and ModuleScope. DeclarationScope is used for any scope that hosts 'var'
68 // declarations. This includes script, module, eval, varblock, and function

429  bool RemoveInnerScope(Scope* inner_scope) {
430  DCHECK_NOT_NULL(inner_scope);
431  if (inner_scope == inner_scope_) {
432  inner_scope_ = inner_scope_->sibling_;
433  return true;
434  }
435  for (Scope* scope = inner_scope_; scope != nullptr;
436  scope = scope->sibling_) {
437  if (scope->sibling_ == inner_scope) {
438  scope->sibling_ = scope->sibling_->sibling_;
439  return true;
440  }
441  }
442  return false;
443  }
