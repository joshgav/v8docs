109 // Defines the arity and the call flags for a JavaScript function call. This is
110 // used as a parameter by JSCallFunction operators.
111 class CallFunctionParameters final {
112  public:
113  CallFunctionParameters(size_t arity, float frequency,
114  VectorSlotPair const& feedback,
115  TailCallMode tail_call_mode,
116  ConvertReceiverMode convert_mode)
117  : bit_field_(ArityField::encode(arity) |
118  ConvertReceiverModeField::encode(convert_mode) |
119  TailCallModeField::encode(tail_call_mode)),
120  frequency_(frequency),
121  feedback_(feedback) {}
122 
123  size_t arity() const { return ArityField::decode(bit_field_); }
124  float frequency() const { return frequency_; }
125  ConvertReceiverMode convert_mode() const {
126  return ConvertReceiverModeField::decode(bit_field_);
127  }
128  TailCallMode tail_call_mode() const {
129  return TailCallModeField::decode(bit_field_);
130  }
131  VectorSlotPair const& feedback() const { return feedback_; }
132 
133  bool operator==(CallFunctionParameters const& that) const {
134  return this->bit_field_ == that.bit_field_ &&
135  this->frequency_ == that.frequency_ &&
136  this->feedback_ == that.feedback_;
137  }
138  bool operator!=(CallFunctionParameters const& that) const {
139  return !(*this == that);
140  }
141 
142  private:
143  friend size_t hash_value(CallFunctionParameters const& p) {
144  return base::hash_combine(p.bit_field_, p.frequency_, p.feedback_);
145  }
146 
147  typedef BitField<size_t, 0, 29> ArityField;
148  typedef BitField<ConvertReceiverMode, 29, 2> ConvertReceiverModeField;
149  typedef BitField<TailCallMode, 31, 1> TailCallModeField;
150 
151  uint32_t const bit_field_;
152  float const frequency_;
153  VectorSlotPair const feedback_;
154 };
