183  static MachineType TypeForRepresentation(const MachineRepresentation& rep,
184  bool isSigned = true) {
185  switch (rep) {
186  case MachineRepresentation::kNone:
187  return MachineType::None();
188  case MachineRepresentation::kBit:
189  return MachineType::Bool();
190  case MachineRepresentation::kWord8:
191  return isSigned ? MachineType::Int8() : MachineType::Uint8();
192  case MachineRepresentation::kWord16:
193  return isSigned ? MachineType::Int16() : MachineType::Uint16();
194  case MachineRepresentation::kWord32:
195  return isSigned ? MachineType::Int32() : MachineType::Uint32();
196  case MachineRepresentation::kWord64:
197  return isSigned ? MachineType::Int64() : MachineType::Uint64();
198  case MachineRepresentation::kFloat32:
199  return MachineType::Float32();
200  case MachineRepresentation::kFloat64:
201  return MachineType::Float64();
202  case MachineRepresentation::kSimd128:
203  return MachineType::Simd128();
204  case MachineRepresentation::kTagged:
205  return MachineType::AnyTagged();
206  case MachineRepresentation::kTaggedSigned:
207  return MachineType::TaggedSigned();
208  case MachineRepresentation::kTaggedPointer:
209  return MachineType::TaggedPointer();
210  default:
211  UNREACHABLE();
212  return MachineType::None();
213  }
214  }

253 // Gets the log2 of the element size in bytes of the machine type.
254 V8_EXPORT_PRIVATE inline int ElementSizeLog2Of(MachineRepresentation rep) {
255  switch (rep) {
256  case MachineRepresentation::kBit:
257  case MachineRepresentation::kWord8:
258  return 0;
259  case MachineRepresentation::kWord16:
260  return 1;
261  case MachineRepresentation::kWord32:
262  case MachineRepresentation::kFloat32:
263  return 2;
264  case MachineRepresentation::kWord64:
265  case MachineRepresentation::kFloat64:
266  return 3;
267  case MachineRepresentation::kSimd128:
268  return 4;
269  case MachineRepresentation::kTaggedSigned:
270  case MachineRepresentation::kTaggedPointer:
271  case MachineRepresentation::kTagged:
272  return kPointerSizeLog2;
273  default:
274  break;
275  }
276  UNREACHABLE();
277  return -1;
278 }
