257 template <typename StaticVisitor>
258 void StaticMarkingVisitor<StaticVisitor>::VisitCodeTarget(Heap* heap,
259  RelocInfo* rinfo) {
260  DCHECK(RelocInfo::IsCodeTarget(rinfo->rmode()));
261  Code* target = Code::GetCodeFromTargetAddress(rinfo->target_address());
262  // Monomorphic ICs are preserved when possible, but need to be flushed
263  // when they might be keeping a Context alive, or when the heap is about
264  // to be serialized.
265  if (FLAG_cleanup_code_caches_at_gc && target->is_inline_cache_stub() &&
266  (heap->isolate()->serializer_enabled() ||
267  target->ic_age() != heap->global_ic_age())) {
268  ICUtility::Clear(heap->isolate(), rinfo->pc(),
269  rinfo->host()->constant_pool());
270  target = Code::GetCodeFromTargetAddress(rinfo->target_address());
271  }
272  Code* host = rinfo->host();
273  heap->mark_compact_collector()->RecordRelocSlot(host, rinfo, target);
274  StaticVisitor::MarkObject(heap, target);
275 }

304 template <typename StaticVisitor>
305 void StaticMarkingVisitor<StaticVisitor>::VisitMap(Map* map,
306  HeapObject* object) {
307  Heap* heap = map->GetHeap();
308  Map* map_object = Map::cast(object);
309 
310  // Clears the cache of ICs related to this map.
311  if (FLAG_cleanup_code_caches_at_gc) {
312  map_object->ClearCodeCache(heap);
313  }
314 
315  // When map collection is enabled we have to mark through map's transitions
316  // and back pointers in a special way to make these links weak.
317  if (map_object->CanTransition()) {
318  MarkMapContents(heap, map_object);
319  } else {
320  StaticVisitor::VisitPointers(
321  heap, object,
322  HeapObject::RawField(object, Map::kPointerFieldsBeginOffset),
323  HeapObject::RawField(object, Map::kPointerFieldsEndOffset));
324  }
325 }

327 template <typename StaticVisitor>
328 void StaticMarkingVisitor<StaticVisitor>::VisitWeakCell(Map* map,
329  HeapObject* object) {
330  Heap* heap = map->GetHeap();
331  WeakCell* weak_cell = reinterpret_cast<WeakCell*>(object);
332  // Enqueue weak cell in linked list of encountered weak collections.
333  // We can ignore weak cells with cleared values because they will always
334  // contain smi zero.
335  if (weak_cell->next_cleared() && !weak_cell->cleared()) {
336  HeapObject* value = HeapObject::cast(weak_cell->value());
337  if (MarkCompactCollector::IsMarked(value)) {
338  // Weak cells with live values are directly processed here to reduce
339  // the processing time of weak cells during the main GC pause.
340  Object** slot = HeapObject::RawField(weak_cell, WeakCell::kValueOffset);
341  map->GetHeap()->mark_compact_collector()->RecordSlot(weak_cell, slot,
342  *slot);
343  } else {
344  // If we do not know about liveness of values of weak cells, we have to
345  // process them when we know the liveness of the whole transitive
346  // closure.
347  weak_cell->set_next(heap->encountered_weak_cells(),
348  UPDATE_WEAK_WRITE_BARRIER);
349  heap->set_encountered_weak_cells(weak_cell);
350  }
351  }
352 }

355 template <typename StaticVisitor>
356 void StaticMarkingVisitor<StaticVisitor>::VisitTransitionArray(
357  Map* map, HeapObject* object) {
358  TransitionArray* array = TransitionArray::cast(object);
359  Heap* heap = array->GetHeap();
360  // Visit strong references.
361  if (array->HasPrototypeTransitions()) {
362  StaticVisitor::VisitPointer(heap, array,
363  array->GetPrototypeTransitionsSlot());
364  }
365  int num_transitions = TransitionArray::NumberOfTransitions(array);
366  for (int i = 0; i < num_transitions; ++i) {
367  StaticVisitor::VisitPointer(heap, array, array->GetKeySlot(i));
368  }
369  // Enqueue the array in linked list of encountered transition arrays if it is
370  // not already in the list.
371  if (array->next_link()->IsUndefined(heap->isolate())) {
372  Heap* heap = map->GetHeap();
373  array->set_next_link(heap->encountered_transition_arrays(),
374  UPDATE_WEAK_WRITE_BARRIER);
375  heap->set_encountered_transition_arrays(array);
376  }
377 }

379 template <typename StaticVisitor>
380 void StaticMarkingVisitor<StaticVisitor>::VisitWeakCollection(
381  Map* map, HeapObject* object) {
382  typedef FlexibleBodyVisitor<StaticVisitor,
383  JSWeakCollection::BodyDescriptorWeak,
384  void> JSWeakCollectionBodyVisitor;
385  Heap* heap = map->GetHeap();
386  JSWeakCollection* weak_collection =
387  reinterpret_cast<JSWeakCollection*>(object);
388 
389  // Enqueue weak collection in linked list of encountered weak collections.
390  if (weak_collection->next() == heap->undefined_value()) {
391  weak_collection->set_next(heap->encountered_weak_collections());
392  heap->set_encountered_weak_collections(weak_collection);
393  }
394 
395  // Skip visiting the backing hash table containing the mappings and the
396  // pointer to the other enqueued weak collections, both are post-processed.
397  JSWeakCollectionBodyVisitor::Visit(map, object);
398 
399  // Partially initialized weak collection is enqueued, but table is ignored.
400  if (!weak_collection->table()->IsHashTable()) return;
401 
402  // Mark the backing hash table without pushing it on the marking stack.
403  Object** slot = HeapObject::RawField(object, JSWeakCollection::kTableOffset);
404  HeapObject* obj = HeapObject::cast(*slot);
405  heap->mark_compact_collector()->RecordSlot(object, slot, obj);
406  StaticVisitor::MarkObjectWithoutPush(heap, obj);
407 }

424 template <typename StaticVisitor>
425 void StaticMarkingVisitor<StaticVisitor>::VisitSharedFunctionInfo(
426  Map* map, HeapObject* object) {
427  Heap* heap = map->GetHeap();
428  SharedFunctionInfo* shared = SharedFunctionInfo::cast(object);
429  if (shared->ic_age() != heap->global_ic_age()) {
430  shared->ResetForNewContext(heap->global_ic_age());
431  }
432  MarkCompactCollector* collector = heap->mark_compact_collector();
433  if (collector->is_code_flushing_enabled()) {
434  if (IsFlushable(heap, shared)) {
435  // This function's code looks flushable. But we have to postpone
436  // the decision until we see all functions that point to the same
437  // SharedFunctionInfo because some of them might be optimized.
438  // That would also make the non-optimized version of the code
439  // non-flushable, because it is required for bailing out from
440  // optimized code.
441  collector->code_flusher()->AddCandidate(shared);
442  // Treat the reference to the code object weakly.
443  VisitSharedFunctionInfoWeakCode(map, object);
444  return;
445  }
446  }
447  VisitSharedFunctionInfoStrongCode(map, object);
448 }

451 template <typename StaticVisitor>
452 void StaticMarkingVisitor<StaticVisitor>::VisitJSFunction(Map* map,
453  HeapObject* object) {
454  Heap* heap = map->GetHeap();
455  JSFunction* function = JSFunction::cast(object);
456  if (FLAG_cleanup_code_caches_at_gc) {
457  function->ClearTypeFeedbackInfoAtGCTime();
458  }
459  MarkCompactCollector* collector = heap->mark_compact_collector();
460  if (collector->is_code_flushing_enabled()) {
461  if (IsFlushable(heap, function)) {
462  // This function's code looks flushable. But we have to postpone
463  // the decision until we see all functions that point to the same
464  // SharedFunctionInfo because some of them might be optimized.
465  // That would also make the non-optimized version of the code
466  // non-flushable, because it is required for bailing out from
467  // optimized code.
468  collector->code_flusher()->AddCandidate(function);
469  // Treat the reference to the code object weakly.
470  VisitJSFunctionWeakCode(map, object);
471  return;
472  } else {
473  // Visit all unoptimized code objects to prevent flushing them.
474  StaticVisitor::MarkObject(heap, function->shared()->code());
475  }
476  }
477  VisitJSFunctionStrongCode(map, object);
478 }

480 template <typename StaticVisitor>
481 void StaticMarkingVisitor<StaticVisitor>::MarkMapContents(Heap* heap,
482  Map* map) {
483  // Since descriptor arrays are potentially shared, ensure that only the
484  // descriptors that belong to this map are marked. The first time a non-empty
485  // descriptor array is marked, its header is also visited. The slot holding
486  // the descriptor array will be implicitly recorded when the pointer fields of
487  // this map are visited. Prototype maps don't keep track of transitions, so
488  // just mark the entire descriptor array.
489  if (!map->is_prototype_map()) {
490  DescriptorArray* descriptors = map->instance_descriptors();
491  if (StaticVisitor::MarkObjectWithoutPush(heap, descriptors) &&
492  descriptors->length() > 0) {
493  StaticVisitor::VisitPointers(heap, descriptors,
494  descriptors->GetFirstElementAddress(),
495  descriptors->GetDescriptorEndSlot(0));
496  }
497  int start = 0;
498  int end = map->NumberOfOwnDescriptors();
499  if (start < end) {
500  StaticVisitor::VisitPointers(heap, descriptors,
501  descriptors->GetDescriptorStartSlot(start),
502  descriptors->GetDescriptorEndSlot(end));
503  }
504  }
505 
506  // Mark the pointer fields of the Map. Since the transitions array has
507  // been marked already, it is fine that one of these fields contains a
508  // pointer to it.
509  StaticVisitor::VisitPointers(
510  heap, map, HeapObject::RawField(map, Map::kPointerFieldsBeginOffset),
511  HeapObject::RawField(map, Map::kPointerFieldsEndOffset));
512 }

522 template <typename StaticVisitor>
523 bool StaticMarkingVisitor<StaticVisitor>::IsFlushable(Heap* heap,
524  JSFunction* function) {
525  SharedFunctionInfo* shared_info = function->shared();
526 
527  // Code is either on stack, in compilation cache or referenced
528  // by optimized version of function.
529  MarkBit code_mark = ObjectMarking::MarkBitFrom(function->code());
530  if (Marking::IsBlackOrGrey(code_mark)) {
531  return false;
532  }
533 
534  // We do not (yet) flush code for optimized functions.
535  if (function->code() != shared_info->code()) {
536  return false;
537  }
538 
539  // Check age of optimized code.
540  if (FLAG_age_code && !function->code()->IsOld()) {
541  return false;
542  }
543 
544  return IsFlushable(heap, shared_info);
545 }

548 template <typename StaticVisitor>
549 bool StaticMarkingVisitor<StaticVisitor>::IsFlushable(
550  Heap* heap, SharedFunctionInfo* shared_info) {
551  // Code is either on stack, in compilation cache or referenced
552  // by optimized version of function.
553  MarkBit code_mark = ObjectMarking::MarkBitFrom(shared_info->code());
554  if (Marking::IsBlackOrGrey(code_mark)) {
555  return false;
556  }
557 
558  // The function must be compiled and have the source code available,
559  // to be able to recompile it in case we need the function again.
560  if (!(shared_info->is_compiled() && HasSourceCode(heap, shared_info))) {
561  return false;
562  }
563 
564  // We never flush code for API functions.
565  if (shared_info->IsApiFunction()) {
566  return false;
567  }
568 
569  // Only flush code for functions.
570  if (shared_info->code()->kind() != Code::FUNCTION) {
571  return false;
572  }
573 
574  // Function must be lazy compilable.
575  if (!shared_info->allows_lazy_compilation()) {
576  return false;
577  }
578 
579  // We do not (yet?) flush code for generator functions, or async functions,
580  // because we don't know if there are still live activations
581  // (generator objects) on the heap.
582  if (IsResumableFunction(shared_info->kind())) {
583  return false;
584  }
585 
586  // If this is a full script wrapped in a function we do not flush the code.
587  if (shared_info->is_toplevel()) {
588  return false;
589  }
590 
591  // The function must be user code.
592  if (!shared_info->IsUserJavaScript()) {
593  return false;
594  }
595 
596  // Maintain debug break slots in the code.
597  if (shared_info->HasDebugCode()) {
598  return false;
599  }
600 
601  // If this is a function initialized with %SetCode then the one-to-one
602  // relation between SharedFunctionInfo and Code is broken.
603  if (shared_info->dont_flush()) {
604  return false;
605  }
606 
607  // Check age of code. If code aging is disabled we never flush.
608  if (!FLAG_age_code || !shared_info->code()->IsOld()) {
609  return false;
610  }
611 
612  return true;
613 }
