37 // AstString, AstValue and AstValueFactory are for storing strings and values
38 // independent of the V8 heap and internalizing them later. During parsing,
39 // AstStrings and AstValues are created and stored outside the heap, in
40 // AstValueFactory. After parsing, the strings and values are internalized
41 // (moved into the V8 heap).

45 class AstString : public ZoneObject {
46  public:
47  explicit AstString(bool is_raw)
48  : next_(nullptr), bit_field_(IsRawStringBits::encode(is_raw)) {}
49 
50  int length() const;
51  bool IsEmpty() const { return length() == 0; }
52 
53  // Puts the string into the V8 heap.
54  void Internalize(Isolate* isolate);
55 
56  // This function can be called after internalizing.
57  V8_INLINE Handle<String> string() const {
58  DCHECK_NOT_NULL(string_);
59  return Handle<String>(string_);
60  }
61 
62  AstString* next() { return next_; }
63  AstString** next_location() { return &next_; }
64 
65  protected:
66  void set_string(Handle<String> string) { string_ = string.location(); }
67  // {string_} is stored as String** instead of a Handle<String> so it can be
68  // stored in a union with {next_}.
69  union {
70  AstString* next_;
71  String** string_;
72  };
73  // Poor-man's virtual dispatch to AstRawString / AstConsString. Takes less
74  // memory.
75  class IsRawStringBits : public BitField<bool, 0, 1> {};
76  int bit_field_;
77 };

82  int length() const {

87  int byte_length() const { return literal_bytes_.length(); }
88 
89  void Internalize(Isolate* isolate);

91  bool AsArrayIndex(uint32_t* index) const;
92 
93  // The string is not null-terminated, use length() to find out the length.
94  const unsigned char* raw_data() const {
95  return literal_bytes_.start();
96  }
97 
98  bool is_one_byte() const { return IsOneByteBits::decode(bit_field_); }
99 
100  bool IsOneByteEqualTo(const char* data) const;
101  uint16_t FirstCharacter() const {
102  if (is_one_byte()) return literal_bytes_[0];
103  const uint16_t* c =
104  reinterpret_cast<const uint16_t*>(literal_bytes_.start());
105  return *c;
106  }
107 
108  // For storing AstRawStrings in a hash map.
109  uint32_t hash() const {
110  return hash_;
111  }
112 
113  private:
114  friend class AstValueFactory;
115  friend class AstRawStringInternalizationKey;
116 
117  AstRawString(bool is_one_byte, const Vector<const byte>& literal_bytes,
118  uint32_t hash)
119  : AstString(true), hash_(hash), literal_bytes_(literal_bytes) {
120  bit_field_ |= IsOneByteBits::encode(is_one_byte);
121  }
122 
123  AstRawString() : AstString(true), hash_(0) {
124  bit_field_ |= IsOneByteBits::encode(true);
125  }
126 
127  class IsOneByteBits : public BitField<bool, IsRawStringBits::kNext, 1> {};
128 
129  uint32_t hash_;
130  // Points to memory owned by Zone.
131  Vector<const byte> literal_bytes_;
132 };

154 // AstValue is either a string, a number, a string array, a boolean, or a
155 // special value (null, undefined, the hole).
156 class AstValue : public ZoneObject {
157  public:
158  bool IsString() const {
159  return type_ == STRING;
160  }
161 
162  bool IsNumber() const {
163  return type_ == NUMBER || type_ == NUMBER_WITH_DOT || type_ == SMI ||
164  type_ == SMI_WITH_DOT;
165  }
166 
167  bool ContainsDot() const {
168  return type_ == NUMBER_WITH_DOT || type_ == SMI_WITH_DOT;
169  }
170 
171  const AstRawString* AsString() const {
172  CHECK_EQ(STRING, type_);
173  return string_;
174  }
175 
176  double AsNumber() const {
177  if (type_ == NUMBER || type_ == NUMBER_WITH_DOT)
178  return number_;
179  if (type_ == SMI || type_ == SMI_WITH_DOT)
180  return smi_;
181  UNREACHABLE();
182  return 0;
183  }
184 
185  Smi* AsSmi() const {
186  CHECK(type_ == SMI || type_ == SMI_WITH_DOT);
187  return Smi::FromInt(smi_);
188  }
189 
190  bool EqualsString(const AstRawString* string) const {
191  return type_ == STRING && string_ == string;
192  }
193 
194  bool IsPropertyName() const;
195 
196  bool BooleanValue() const;
197 
198  bool IsSmi() const { return type_ == SMI || type_ == SMI_WITH_DOT; }
199  bool IsFalse() const { return type_ == BOOLEAN && !bool_; }
200  bool IsTrue() const { return type_ == BOOLEAN && bool_; }
201  bool IsUndefined() const { return type_ == UNDEFINED; }
202  bool IsTheHole() const { return type_ == THE_HOLE; }
203  bool IsNull() const { return type_ == NULL_TYPE; }
204 
205  void Internalize(Isolate* isolate);
206 
207  // Can be called after Internalize has been called.
208  V8_INLINE Handle<Object> value() const {
209  if (type_ == STRING) {
210  return string_->string();
211  }
212  DCHECK_NOT_NULL(value_);
213  return Handle<Object>(value_);
214  }
215  AstValue* next() const { return next_; }
216  void set_next(AstValue* next) { next_ = next; }
217 
218  private:
219  void set_value(Handle<Object> object) { value_ = object.location(); }
220  friend class AstValueFactory;

223  STRING,
224  SYMBOL,
225  NUMBER,

235  explicit AstValue(const AstRawString* s) : type_(STRING), next_(nullptr) {
236  string_ = s;
237  }

239  explicit AstValue(const char* name) : type_(SYMBOL), next_(nullptr) {
240  symbol_name_ = name;

243  explicit AstValue(double n, bool with_dot) : next_(nullptr) {
244  int int_value;
245  if (DoubleToSmiInteger(n, &int_value)) {
246  type_ = with_dot ? SMI_WITH_DOT : SMI;
247  smi_ = int_value;
248  } else {
249  type_ = with_dot ? NUMBER_WITH_DOT : NUMBER;
250  number_ = n;
251  }
252  }
253 
254  AstValue(Type t, int i) : type_(t), next_(nullptr) {
255  DCHECK(type_ == SMI);
256  smi_ = i;
257  }
258 
259  explicit AstValue(bool b) : type_(BOOLEAN), next_(nullptr) { bool_ = b; }
260 
261  explicit AstValue(Type t) : type_(t), next_(nullptr) {
262  DCHECK(t == NULL_TYPE || t == UNDEFINED || t == THE_HOLE);
263  }
264 
265  Type type_;
266 
267  // {value_} is stored as Object** instead of a Handle<Object> so it can be
268  // stored in a union with {next_}.
269  union {
270  Object** value_; // if internalized
271  AstValue* next_; // if !internalized
272  };
273 
274  // Uninternalized value.
275  union {
276  const AstRawString* string_;
277  double number_;
278  int smi_;
279  bool bool_;
280  const char* symbol_name_;
281  };
