336  // Push two registers. Pushes leftmost register first (to highest address).
337  void Push(Register src1, Register src2, Condition cond = al) {
338  if (src1.code() > src2.code()) {
339  stm(db_w, sp, src1.bit() | src2.bit(), cond);
340  } else {
341  str(src1, MemOperand(sp, 4, NegPreIndex), cond);
342  str(src2, MemOperand(sp, 4, NegPreIndex), cond);
343  }
344  }

346  // Push three registers. Pushes leftmost register first (to highest address).
347  void Push(Register src1, Register src2, Register src3, Condition cond = al) {
348  if (src1.code() > src2.code()) {
349  if (src2.code() > src3.code()) {
350  stm(db_w, sp, src1.bit() | src2.bit() | src3.bit(), cond);
351  } else {
352  stm(db_w, sp, src1.bit() | src2.bit(), cond);
353  str(src3, MemOperand(sp, 4, NegPreIndex), cond);
354  }
355  } else {
356  str(src1, MemOperand(sp, 4, NegPreIndex), cond);
357  Push(src2, src3, cond);
358  }
359  }

750  static inline int GetCodeMarker(Instr instr) {
751  int dst_reg_offset = 12;
752  int dst_mask = 0xf << dst_reg_offset;
753  int src_mask = 0xf;
754  int dst_reg = (instr & dst_mask) >> dst_reg_offset;
755  int src_reg = instr & src_mask;
756  uint32_t non_register_mask = ~(dst_mask | src_mask);
757  uint32_t mov_mask = al | 13 << 21;
758 
759  // Return <n> if we have a mov rn rn, else return -1.
760  int type = ((instr & non_register_mask) == mov_mask) &&
761  (dst_reg == src_reg) &&
762  (FIRST_IC_MARKER <= dst_reg) && (dst_reg < LAST_CODE_MARKER)
763  ? src_reg
764  : -1;
765  DCHECK((type == -1) ||
766  ((FIRST_IC_MARKER <= type) && (type < LAST_CODE_MARKER)));
767  return type;
768  }
