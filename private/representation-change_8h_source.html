103 // The {UseInfo} class is used to describe a use of an input of a node.
104 //
105 // This information is used in two different ways, based on the phase:
106 //
107 // 1. During propagation, the use info is used to inform the input node
108 // about what part of the input is used (we call this truncation) and what
109 // is the preferred representation. For conversions that will require
110 // checks, we also keep track of whether a minus zero check is needed.
111 //
112 // 2. During lowering, the use info is used to properly convert the input
113 // to the preferred representation. The preferred representation might be
114 // insufficient to do the conversion (e.g. word32->float64 conv), so we also
115 // need the signedness information to produce the correct value.
116 class UseInfo {
117  public:
118  UseInfo(MachineRepresentation representation, Truncation truncation,
119  TypeCheckKind type_check = TypeCheckKind::kNone,
120  CheckForMinusZeroMode minus_zero_check =
121  CheckForMinusZeroMode::kCheckForMinusZero)
122  : representation_(representation),
123  truncation_(truncation),
124  type_check_(type_check),
125  minus_zero_check_(minus_zero_check) {}
126  static UseInfo TruncatingWord32() {
127  return UseInfo(MachineRepresentation::kWord32, Truncation::Word32());
128  }
129  static UseInfo TruncatingWord64() {
130  return UseInfo(MachineRepresentation::kWord64, Truncation::Word64());
131  }
132  static UseInfo Bool() {
133  return UseInfo(MachineRepresentation::kBit, Truncation::Bool());
134  }
135  static UseInfo Float32() {
136  return UseInfo(MachineRepresentation::kFloat32, Truncation::Any());
137  }
138  static UseInfo TruncatingFloat64() {
139  return UseInfo(MachineRepresentation::kFloat64, Truncation::Float64());
140  }
141  static UseInfo PointerInt() {
142  return kPointerSize == 4 ? TruncatingWord32() : TruncatingWord64();
143  }
144  static UseInfo AnyTagged() {
145  return UseInfo(MachineRepresentation::kTagged, Truncation::Any());
146  }
147  static UseInfo TaggedSigned() {
148  return UseInfo(MachineRepresentation::kTaggedSigned, Truncation::Any());
149  }
150  static UseInfo TaggedPointer() {
151  return UseInfo(MachineRepresentation::kTaggedPointer, Truncation::Any());
152  }
153 
154  // Possibly deoptimizing conversions.
155  static UseInfo CheckedSignedSmallAsTaggedSigned() {
156  return UseInfo(MachineRepresentation::kTaggedSigned, Truncation::Any(),
157  TypeCheckKind::kSignedSmall);
158  }
159  static UseInfo CheckedSignedSmallAsWord32(
160  CheckForMinusZeroMode minus_zero_mode =
161  CheckForMinusZeroMode::kCheckForMinusZero) {
162  return UseInfo(MachineRepresentation::kWord32, Truncation::Any(),
163  TypeCheckKind::kSignedSmall, minus_zero_mode);
164  }
165  static UseInfo CheckedSigned32AsWord32(
166  CheckForMinusZeroMode minus_zero_mode =
167  CheckForMinusZeroMode::kCheckForMinusZero) {
168  return UseInfo(MachineRepresentation::kWord32, Truncation::Any(),
169  TypeCheckKind::kSigned32, minus_zero_mode);
170  }
171  static UseInfo CheckedNumberAsFloat64() {
172  return UseInfo(MachineRepresentation::kFloat64, Truncation::Float64(),
173  TypeCheckKind::kNumber);
174  }
175  static UseInfo CheckedNumberAsWord32() {
176  return UseInfo(MachineRepresentation::kWord32, Truncation::Word32(),
177  TypeCheckKind::kNumber);
178  }
179  static UseInfo CheckedNumberOrOddballAsFloat64() {
180  return UseInfo(MachineRepresentation::kFloat64, Truncation::Any(),
181  TypeCheckKind::kNumberOrOddball);
182  }
183  static UseInfo CheckedNumberOrOddballAsWord32() {
184  return UseInfo(MachineRepresentation::kWord32, Truncation::Word32(),
185  TypeCheckKind::kNumberOrOddball);
186  }
187 
188  // Undetermined representation.
189  static UseInfo Any() {
190  return UseInfo(MachineRepresentation::kNone, Truncation::Any());
191  }
192  static UseInfo AnyTruncatingToBool() {
193  return UseInfo(MachineRepresentation::kNone, Truncation::Bool());
194  }

196  // Value not used.
