177  // Find any implicitly indirect exports and make them explicit.
178  //
179  // An explicitly indirect export is an export entry arising from an export
180  // statement of the following form:
181  // export {a as c} from "X";
182  // An implicitly indirect export corresponds to
183  // export {b as c};
184  // in the presence of an import statement of the form
185  // import {a as b} from "X";
186  // This function finds such implicitly indirect export entries and rewrites
187  // them by filling in the import name and module request, as well as nulling
188  // out the local name. Effectively, it turns
189  // import {a as b} from "X"; export {b as c};
190  // into:
191  // import {a as b} from "X"; export {a as c} from "X";
192  // (The import entry is never deleted.)
193  void MakeIndirectExportsExplicit(Zone* zone);

195  int AddModuleRequest(const AstRawString* specifier) {
196  DCHECK_NOT_NULL(specifier);
197  auto it = module_requests_
198  .insert(std::make_pair(specifier, module_requests_.size()))
199  .first;
200  return it->second;
